    <script>
        const itemIds = ['location', 'plan', 'material', 'common', 'facade', 'access', 'density', 'price'];
        
        const icons = {
            'location': 'fa-map-marker-alt',
            'plan': 'fa-drafting-compass',
            'material': 'fa-gem',
            'common': 'fa-people-arrows',
            'facade': 'fa-building',
            'access': 'fa-road',
            'density': 'fa-city',
            'price': 'fa-tags'
        };

        window.onload = function() {
            const d = new Date();
            const formatter = new Intl.DateTimeFormat('fa-IR', { year: 'numeric', month: 'numeric', day: 'numeric' });
            document.getElementById('date-now').innerText = formatter.format(d);
            calculateAverage();
        };

        function updateVal(id) {
            const val = document.getElementById(id).value;
            document.getElementById(val-${id}).innerText = val;
            calculateAverage();
        }

        function calculateAverage() {
            let total = 0;
            itemIds.forEach(id => {
                total += parseInt(document.getElementById(id).value);
            });
            const avg = (total / itemIds.length).toFixed(1);
            document.getElementById('manualTotal').value = avg;
        }

        function toggleGuide() {
            const modal = document.getElementById('codingGuideModal');
            if (modal.classList.contains('hidden')) {
                modal.classList.remove('hidden');
            } else {
                modal.classList.add('hidden');
            }
        }

        function generatePreview() {
            const container = document.getElementById('grid-container');
            container.innerHTML = '';

            itemIds.forEach(id => {
                const val = document.getElementById(id).value;
                const labelText = document.getElementById('label-' + id).value;
                const iconClass = icons[id]; 
                
                const div = document.createElement('div');
                div.className = flex flex-col p-1 bg-gray-50 rounded border border-blue-100 relative shadow-[0_0_0_1px_rgba(250,204,21,0.2)];

                div.innerHTML = `
                    <div class="flex items-center gap-1 mb-0.5">
                        <i class="fas ${iconClass} text-[7px] text-blue-400"></i>
